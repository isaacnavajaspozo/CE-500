# 💃 dentro del samba creo una carpeta por usuario y le agrego una base de datos de keepassxc por usuario

# creo la base de datos
keepassxc-cli db-create mydatabase.kdbx --set-password
> **********

# abro la base d edatos
keepassxc-cli open mydatabase.kdbx
