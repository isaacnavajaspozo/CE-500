[⛩️CREAR ALIAS EN SERVIDOR PRINCIPAL]::
# 🧑‍🎓 Creo un alias para poder acceder directamente desde cualquier parte del servidor principal
# recuerdo de que el script estaba en la siguiente ruta /KIRIBAKO/lxc-manager.sh
ls /KIRIBAKO | grep lxc-manager.sh

# Entro a bashrc
vim ~/.bashrc
--------------------------------------------------------------------------------------------------------------------------
## alias para ejecutar el script principal de las máquinas
alias _lxc-manager="/KIRIBAKO/lxc-manager.sh"
--------------------------------------------------------------------------------------------------------------------------

# actualizo los cambios 
source ~/.bashrc

# ejecuto script
_lxc-manager
