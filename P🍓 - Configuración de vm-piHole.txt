# 🧠 Pi Hole sirve para bloquear lso anuncios web.
# 🐧 Lo voy a utilizar para resolver dudas que me aparezcan y en especial comprender la arquitectura de diseño de los servicios utilizados en el momento de la instalación.
#=========================================================================================================================
[🍓 KIRIBAKO]::
# documentación oficial   : https://ollama.com/

#=========================================================================================================================
[🍓 INSTALACIÓN dependencias]::
# instala y habilita el servidor web lighttpd (usado por Pi-hole para la interfaz web)
sudo apt install lighttpd -y
sudo systemctl enable --now lighttpd

# instala curl si no está instalado
sudo apt install curl -y

# necesidades web
sudo apt install php-cgi php-common php -y
sudo lighty-enable-mod fastcgi
sudo lighty-enable-mod fastcgi-php
sudo systemctl restart lighttpd


#=========================================================================================================================
[🍓 CONFIGURACIÓN]::
# compruebo 
sudo systemctl status lighttpd

# actualiza el sistema
sudo apt update && sudo apt upgrade -y

# ejecuta el instalador oficial
curl -sSL https://install.pi-hole.net | bash

# configuración web
http://<IP-Local>/admin


#=========================================================================================================================
[🍓 COMPROBACIÓN]::
# verifica funcionamiento
pihole status

# modificar contraseña 
pihole setpassword

# generar listas de bloqueo desde la web en Group Management > Adlists.
# generar listas de bloqueo desde terminal:
pihole -g                                # Regenera las listas
pihole -b dominio                        # Añade dominio a lista negra
pihole -w dominio                        # Añade dominio a lista blanca
pihole -t                                # comprobar listas en tiempo real
